VBScript and ASP Code: Demonstrating the Buffer Property of the Response Object

<%@ LANGUAGE="VBScript" %>
<% Option Explicit %>
<%
Response.Buffer = True


Public Sub basDelaySec(Seconds) 

    Dim dtmDate 

    dtmDate = DateAdd(&quot;s&quot;, Seconds, Now) 
     
    Do 
    Loop Until Now &gt; dtmDate 

End Sub
