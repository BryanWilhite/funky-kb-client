Perl: Returning the Last Five Errors from a Server Log

#!/path/to/perl
open(CMDPIPE, "tail -5 /path/to/server/error/log |");
print "Content-type: text/html\\n\\n";
print "<HTML>\\n";
print "<PRE>\\n";
print <CMDPIPE>;
close (CMDPIPE);
print "</PRE>\\n";
print "</HTML>\\n";
