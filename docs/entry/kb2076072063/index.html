Apache HTTP Server 2.x: Force HTTPS by Port; .htaccess

RewriteEngine on


#Force HTTPS by port:
RewriteCond %{SERVER_PORT} !^443$
RewriteRule (.*) https://%{SERVER_NAME}%{REQUEST_URI} [R=301,L]
