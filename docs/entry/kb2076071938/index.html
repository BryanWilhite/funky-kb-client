SQL Server 2005 t-SQL: The Use of the nodes() Method for SQL Types; Converting XML Data into SQL Types

DECLARE @xml xml


SET @xml = '
<Root>
<Item>one</Item>
<Item>two</Item>
<Item>three</Item>
<Item>four</Item>
<Item>five</Item>
</Root>
'


SELECT
Temp.Items.value('.','varchar(16)') AS [Items]
FROM
@xml.nodes('//Item') AS Temp(Items)
