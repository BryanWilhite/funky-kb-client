DECLARE @xml xml

SET @xml = ' <Root> <Item>one</Item> <Item>two</Item> <Item>three</Item> <Item>four</Item> <Item>five</Item> </Root> '

SELECT Temp.Items.value('.','varchar(16)') AS [Items] FROM @xml.nodes('//Item') AS Temp(Items)
