SQL Server 2005 t-SQL: Basic XQuery Example

DECLARE @x xml


SET @x = '
<root>
<myElement myAttribute="attrValue">The element value.</myElement>
</root>
'


SELECT
@x.query('//myElement/node()') as [element value]
,   @x.query('data(//myElement)') as [element value]
,   @x.query('data(//myElement/@myAttribute)') as [elment attribute]
