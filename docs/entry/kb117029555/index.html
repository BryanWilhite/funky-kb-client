Perl: Listing the Environment Variables Exposed to a CGI Session

#!/path/to/perl
print "Content-type: text/html\\n\\n";
print "<HTML>\\n";
print "<PRE>\\n";
@keys = keys %ENV;
@values = values %ENV;
foreach $key (sort(keys %ENV)) { print "$key = $ENV{$key}\\n" };
print "</PRE>\\n";
print "</HTML>\\n";
