VBScript and ASP Code Snippet: LOGON_USER Split(); vLoginDomainName, vLoginUserName 

Dim vLogin : vLogin = Request.ServerVariables("LOGON_USER")


Dim vLoginDomainName, vLoginUserName
If InStr(1,vLogin,"\\",vbTextCompare) > 0 Then
vLoginDomainName = Split(vLogin,"\\",-1,vbTextCompare)(0)
vLoginUserName = Split(vLogin,"\\",-1,vbTextCompare)(1)
End If
