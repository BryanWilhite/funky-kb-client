C# Code: DatumStateEnum

namespace AppXCommon.Enum
{
/// <summary>
/// Defines client-side states for data display.
/// </summary>
public enum DatumStateEnum
{
/// <summary>
/// Data is null.
/// </summary>
Unknown,


    /// &lt;summary&gt;
    /// Data is available.
    /// &lt;/summary&gt;
    Available,

    /// &lt;summary&gt;
    /// Data is available and indicates an downward trend.
    /// &lt;/summary&gt;
    AvailableAndDown,

    /// &lt;summary&gt;
    /// Data is available and indicates an upward trend.
    /// &lt;/summary&gt;
    AvailableAndUp,

    /// &lt;summary&gt;
    /// Data is not applicable for display.
    /// &lt;/summary&gt;
    NotApplicable,

    /// &lt;summary&gt;
    /// Data is not available.
    /// &lt;/summary&gt;
    NotAvailable
}



}
