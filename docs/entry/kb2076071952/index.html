LINQ to XML: Using System.Linq.Enumerable to Generate XML; Code from Mike Taulty, Microsoft UK

/*
using System;
using System.Linq;
using System.Xml.Linq;
*/


var numbers =


from n in Enumerable.Range( 1, 20 )

let attributes =
    new Object[]{
        new XAttribute( &quot;value&quot;, n ),
        new XAttribute( &quot;squared&quot;, n * n ) }

select new XElement( &quot;number&quot;, attributes );



XElement document = new XElement( "numbers", numbers );


Console.WriteLine( document.ToString() );
