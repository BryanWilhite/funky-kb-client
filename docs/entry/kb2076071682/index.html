C# and ASP.NET Code Snippet: RequestFormElement()

/// <summary>
/// Returns the ASP.NET-generated key
/// for a <see cref="System.Web.UI.Control"/>
/// in a <see cref="System.Web.HttpRequest.Form"/>
/// collection.
/// </summary>
/// <param name="Control">
/// The <see cref="System.Web.UI.Control"/> to find.
/// </param>
public static string RequestFormElement(System.Web.UI.Control Control)
{
string str =
Control.ClientID.Replace(String.Format("_{0}",Control.ID),
String.Format(":{0}",Control.ID));
return str;
}
