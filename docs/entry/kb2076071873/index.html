C# 2.0 Code: Converting from IDataParameter to DbParameter[] with Array.ConvertAll<>() and an Anonymous Delegate

private static DbParameter[] ConvertAllToDbParameter
( IDataParameter[] p_idata )
{
DbParameter[] p =
Array.ConvertAll<IDataParameter, DbParameter>( p_idata,


       delegate( IDataParameter from )
       {
           DbParameter to = from as DbParameter;
           return to;
       }

   );

return p;



}
