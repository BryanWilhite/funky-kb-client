bash script: a script that generates symbolic links based on subdirectory names in a target folder

#!/bin/bash


echo "Removing any existing symbolic links..."
rm /opt/lampp/htdocs/dev_*


for folder in find ~/webRoot/ -name 'dev_*' -type d
do
echo "Linking to $folder..."
ln -s $folder /opt/lampp/htdocs/basename $folder
done


exit 0
