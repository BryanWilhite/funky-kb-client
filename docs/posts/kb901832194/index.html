Dim lngCount As Long
Dim strChar As String
Dim strRet As String
Dim strCmp As String

If InclDec Then
    strCmp = &quot;1234567890.&quot;
Else
    strCmp = &quot;1234567890&quot;
End If

    strChar = VBA.Mid$(str, lngCount, 1)
    If VBA.InStr(strCmp, strChar) &gt; 0 Then
        strRet = strRet & strChar
    End If
Next
