namespace AppXCommon.Enum { /// <summary> /// Defines client-side states for data display. /// </summary> public enum DatumStateEnum { /// <summary> /// Data is null. /// </summary> Unknown,

    /// Data is available.
    /// &lt;/summary&gt;
    Available,

    /// &lt;/summary&gt;

    /// &lt;/summary&gt;

    /// &lt;/summary&gt;
    NotAvailable
