private static DbParameter[] ConvertAllToDbParameter
    ( IDataParameter[] p_idata )
{
    DbParameter[] p =
        Array.ConvertAll&lt;IDataParameter, DbParameter&gt;( p_idata,

           delegate( IDataParameter from )
           {
               DbParameter to = from as DbParameter;
               return to;
           }

       );

    return p;
}
