Public Sub basDelaySec(ByVal Seconds As Long)

    Dim dtmDate as Date

    dtmDate = DateAdd(&quot;s&quot;, Seconds, Now)
    
    Do
        DoEvents
    Loop Until Now &gt; dtmDate

End Sub
