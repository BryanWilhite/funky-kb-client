SELECT
    p.*
FROM
    OPENROWSET
    (
        'SQLOLEDB'
    ,   'Trusted_Connection=yes;
         Integrated Security=SSPI;
         DataSource=MyServer;Initial_Catalog=pubs;'
    ,   '
         SELECT * FROM 
         pubs.dbo.authors
         ORDER BY au_lname, au_fname'
    ) AS p
