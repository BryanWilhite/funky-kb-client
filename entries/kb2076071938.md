DECLARE @xml xml

SET @xml = '
    &lt;Root&gt;
        &lt;Item&gt;one&lt;/Item&gt;
        &lt;Item&gt;two&lt;/Item&gt;
        &lt;Item&gt;three&lt;/Item&gt;
        &lt;Item&gt;four&lt;/Item&gt;
        &lt;Item&gt;five&lt;/Item&gt;
    &lt;/Root&gt;
'

SELECT
    Temp.Items.value('.','varchar(16)') AS [Items]
FROM
    @xml.nodes('//Item') AS Temp(Items)
