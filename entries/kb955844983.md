&lt;% Option Explicit %&gt;
&lt;%
    Response.Buffer = True
    
    Public Sub basDelaySec(Seconds) 
    
        Dim dtmDate 
    
        dtmDate = DateAdd(&quot;s&quot;, Seconds, Now) 
         
        Do 
        Loop Until Now &gt; dtmDate 
    
    End Sub
